\makeatletter
\def\@maketitle{
	\newpage
	\vspace*{-\topskip}
	\begingroup\centering
	\let \footnote \thanks
	\hrule height \z@
	{\fontsize{16}{19.2} \@title \par}
	\vskip 0.25em 
	{\large
		\lineskip 0.5em 
		\begin{tabular}[t]{c}
			\@author
		\end{tabular}\par}
	\vskip 0.25em 
	{\large \monthyeardate\today}
	\par\endgroup
	\vskip 1em
}
\makeatother
